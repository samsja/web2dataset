class Document:
    def __init__(
        self,
        origin: str,
        timezone: ZoneInfo = ZoneInfo("Europe/Paris"),
        image_url: Optional[str] = None,
    ):
        self.date = datetime.now(timezone)
        self.origin = origin
        self.image_url = image_url
        self.uuid = uuid.uuid1()

    def __repr__(self):
        return jsons.dumps(self)
